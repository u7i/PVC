149 void PVC::App::ControllerLogic::selectPlayerFromAvailable() {
150  auto select = [&](int index, Player *player){
151  this->currentIndex = index;
152  this->currentServiceName = player->getData().serviceName;
153 
154  this->currentPlayerUpdated();
155  };
156 
157  // Set priority to playing player, but if no players playing - we select the last
158  int lowPriority = -1;
159 
160  for (int c = this->manager->getAvailablePlayers().size() - 1; c >= 0; --c) {
161  Player *player = this->getAvailablePlayerByIndex(c);
162 
163  if (player->getData().status != PlaybackStatus::Stopped) {
164  if (player->getData().status == PlaybackStatus::Playing) { // Immediately select player
165  select(c, player);
166  return;
167  }
168 
169  lowPriority = c;
170  }
171  }
172 
173  // Try to select low-priority player
174  if (lowPriority != -1){
175  select(lowPriority, this->getAvailablePlayerByIndex(lowPriority));
176  return;
177  }
178 
179  // If we found nothing - set none player ( = set currentIndex to -1 )
180  this->selectNonePlayer();
181 }
